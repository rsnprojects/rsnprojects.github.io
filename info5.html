int redLED = 8;      // Pin for red LED
int blueLED = 9;     // Pin for blue LED

void setup() {
  pinMode(redLED, OUTPUT);
  pinMode(blueLED, OUTPUT);
}

void loop() {
  int delayTime = 100; // Delay time between flashes

  // Blink red LED twice
  for (int i = 0; i < 2; i++) {
    digitalWrite(redLED, HIGH);
    digitalWrite(blueLED, LOW);
    delay(delayTime);
    digitalWrite(redLED, LOW);
    delay(delayTime);
  }

  // Blink blue LED twice
  for (int i = 0; i < 2; i++) {
    digitalWrite(redLED, LOW);
    digitalWrite(blueLED, HIGH);
    delay(delayTime);
    digitalWrite(blueLED, LOW);
    delay(delayTime);
  }
}
